DEVELOPER API

Build Something Great

Get started with just a few lines of code

Create User

// Import the ShockDK
import { Pub } from '@shocknet/sdk';

// Create a new user with Nostr auth
Pub.createUser({
  npub: 'npub1...'
}).then(user => {
  console.log('User created!', user.id);
});

Create Invoice

// Generate a Lightning invoice ⚡
Pub.createInvoice({
  amount: 1000,
  memo: 'Coffee payment'
}).then(invoice => {
  console.log(invoice.bolt11);
});

Handle Offers

// Configure webhook for offer callbacks
app.post('/webhook/offers', (req, res) => {
  const { offerId, amount, paid } = req.body;
  
  console.log('Offer status: 💰', {
    offerId,
    amount,
    paid
  });
});

Send Payment

// Pay a Lightning invoice ⚡
Pub.sendPayment({
  invoice: 'lnbc...',
  amount: 1000
}).then(payment => {
  console.log('Paid!', payment.hash);
});
